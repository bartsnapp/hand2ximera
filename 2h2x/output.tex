\begin{example}
Compute 
\[
\lim_{x\to 3} \frac{x^2+6x-5}{x-3}.
\]
\begin{explanation}
Note that when $x \neq 3$, we may write:
\[
\frac{x^2+6x-5}{x-3} = \frac{(x-3)\left(\answer{x+9}\right)}{x-3} = \answer{x+9}.
\]
Since the definition of a limit assumes \wordChoice{\choice{continuity} \choice[correct]{continuity} \choice{continuity}} when $x \neq 3$, the limit exists:
\[
\lim_{x \to 3} \frac{x^2+6x-5}{x-3} = \lim_{x \to 3} \answer{x+9} = \answer{12}.
\]
\end{explanation}
\end{example}
